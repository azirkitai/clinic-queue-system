<script>
  // Auto-scale 1920×1080 ke mana-mana skrin, border luar kekal hitam
  (function(){
    const W=1920, H=1080, stage=document.getElementById('stage');
    function fit(){
      const s = Math.min(window.innerWidth/W, window.innerHeight/H);
      stage.style.transform = `translate(-50%,-50%) scale(${s})`;
    }
    window.addEventListener('resize', fit);
    window.addEventListener('orientationchange', fit);
    fit();

    // (Opsyen) “unlock” audio sekali bila user interact
    window.addEventListener('click', onceUnlock, { once:true });
    window.addEventListener('keydown', onceUnlock, { once:true });
    function onceUnlock(){
      const a = new Audio(); a.muted=true; a.play().finally(()=>a.pause());
    }

    // (Opsyen) Toggle debug frame: tekan "D"
    document.addEventListener('keydown', e=>{
      if(e.key.toLowerCase()==='d'){
        stage.style.outline = stage.style.outline ? '' : '6px solid rgba(255,0,102,.5)';
      }
    });
  })();
</script>

Cara guna: letak SEMUA UI kau (logo, “CALLING”, senarai giliran, poster) di dalam #stage sahaja. Jangan ubah width:1920px; height:1080px—itu “kanvas”. Luarannya akan jadi border hitam ikut TV.

2) Integrasi cepat dengan projek sedia ada

Jika kau dah ada page TV:

Balut content asal dengan <div id="stage"><div class="wrap">…</div></div>.

Tambah CSS #stage{...} dan JS fit() di atas.

Buang height: 100vh/px pada container lama—biar #stage yang jaga scale.

React? Buat komponen Stage1920 dan render children dalamnya; panggil fit() dalam useEffect.

// React contoh ringkas
function Stage1920({children}) {
  const ref = React.useRef(null);
  React.useEffect(()=>{
    const W=1920,H=1080,stage=ref.current;
    const fit=()=>stage.style.transform=`translate(-50%,-50%) scale(${Math.min(innerWidth/W, innerHeight/H)})`;
    addEventListener('resize',fit); fit(); return ()=>removeEventListener('resize',fit);
  },[]);
  return (
    <div id="stage" ref={ref} style={{
      position:'fixed',left:'50%',top:'50%',width:1920,height:1080,
      transform:'translate(-50%,-50%) scale(1)',transformOrigin:'top left',
      background:'#0f1d3a',color:'#fff',overflow:'hidden'
    }}>
      <div className="wrap">{children}</div>
    </div>
  );
}
